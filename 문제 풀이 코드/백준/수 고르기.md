  .readFileSync(0, "utf-8")
  .toString()
  .trim()

  .shift()
  .split(" ")
  .map((v) => +v);

  let [s, e] = [0, 0];
  while (e < n) {

      ans = Math.min(ans, arr[e] - arr[s]);
      ++s;
    } else {
      ++e;
    }
  }
};

twoPointer();
